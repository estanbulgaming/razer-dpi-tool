# FPS Mode - Optimized for footsteps (CS2, Valorant, etc.)
# EqualizerAPO configuration
# Reduces bass, boosts mid frequencies where footsteps are

Preamp: -3 dB

# Cut low frequencies (reduces rumble, explosions masking footsteps)
Filter 1: ON LP Fc 60 Hz Gain -6 dB
Filter 2: ON PK Fc 100 Hz Gain -4 dB Q 1.0

# Slight cut in low-mids
Filter 3: ON PK Fc 250 Hz Gain -2 dB Q 1.0

# Boost presence frequencies (footsteps, reloads, weapon switches)
Filter 4: ON PK Fc 1000 Hz Gain 3 dB Q 1.5
Filter 5: ON PK Fc 2000 Hz Gain 4 dB Q 1.5
Filter 6: ON PK Fc 3500 Hz Gain 3 dB Q 1.5

# Slight boost for detail (enemy positioning cues)
Filter 7: ON PK Fc 6000 Hz Gain 2 dB Q 1.0

# Roll off harsh highs
Filter 8: ON HP Fc 10000 Hz Gain -2 dB
